46    public void mostrarTabla(){
47        for (int i = 0; i < this.tabla.length; i++) {
48            for (int j = 0; j < this.tabla[i].length; j++) {

71    // SE INSERTA EL SIMBOLO EN LA TABLA
72    public void insertarEn(int columna){
73        
74        for(int i = tabla.length - 1; i >= 0 ; i--) {
75            
76            if(tabla[i][columna] == vacio){
77                if (turno){
78                tabla[i][columna] = j1;
79                }else{
80                tabla[i][columna] = j2;

88    public boolean validarPos(int fila, int columna){
89        if (fila >= 0 && fila < tabla.length && columna >= 0 && columna < tabla[0].length){
90            return true;
91        }else{
92            return false;

96    // indica si la tabla esta llena
97    public boolean tablaLlena(){
98        for (int i = 0; i < tabla.length; i++) {
99            for (int j = 0; j < tabla[i].length; j++) {
100                // comprobacion de casilla actual
101                if (tabla[i][j] == vacio){
102                    return false;
103                }
104            }
105        }
106        // devuelve verdadero si la tabla esta llena
107        return true;
108    }

110    // COMPROBACION DE GANADOR EN LA COLUMNA
111    private char columna(){
112        // recorre cada columna
113        for (int j = 0; j < tabla.length; j++) {
114            for (int i = 0; i <= 2; i++) {
115                char simbolo = tabla[i][j];
116                if(simbolo != vacio 
117                   && simbolo == tabla[i+1][j]
118                   && simbolo == tabla[i+2][j]
119                   && simbolo == tabla[i+3][j]){
120                    return simbolo;
121                }
122            }
123        }
124        return vacio;
125    }

129    private char fila(){
130        char simbolo;
131        for (int i = 0; i < 6; i++) {
132            for (int j = 0; j < 4; j++) {
133                simbolo = tabla[i][j];
134                if(simbolo != vacio 
135                   && simbolo == tabla[i][j+1]
136                   && simbolo == tabla[i][j+2]
137                   && simbolo == tabla[i][j+3]){
138                    return simbolo;
139                }
140            }
141        }
142        return vacio;
143    }

146    // COMPROBACION DE GANADOR EN LA DIAGONAL
147    private char diagonal(){
148        char simbolo;
149        
150        // diagonal principal (\)
151        for (int i = 0; i <= 2; i++) {
152            for (int j = 0; j <= 3; j++) {
153                simbolo = tabla[i][j];
154                if(simbolo != vacio
155                   && simbolo == tabla[i+1][j+1]
156                   && simbolo == tabla[i+2][j+2]
157                   && simbolo == tabla[i+3][j+3]){
158                    return simbolo;
159                }
160            }
161        }
162        
163        // diagonal inversa (/)
164        for (int i = 3; i <= 5; i++) {
165            for (int j = 0; j <= 3; j++) {
166                simbolo = tabla[i][j];
167                if(simbolo != vacio
168                   && simbolo == tabla[i-1][j+1]
169                   && simbolo == tabla[i-2][j+2]
170                   && simbolo == tabla[i-3][j+3]){
171                    return simbolo;
172                }
173            }
174        }
175        return vacio;
176    }

178    public boolean finPartida(){
179        if (tablaLlena() 
180            || fila() != vacio 
181            || columna() != vacio 
182            || diagonal() != vacio){
183            return true;
184        }
185        return false;
186    }

188    public void ganador(){
189        char simboloGanador = vacio;
190        if (fila() != vacio){
191            simboloGanador = fila();
192        }else if (columna() != vacio){
193            simboloGanador = columna();
194        }else if (diagonal() != vacio){
195            simboloGanador = diagonal();
196        }
197        
198        if (simboloGanador == j1){
199            System.out.println("Ha ganado el jugador 1: " + j1);
200        }
201        else if (simboloGanador == j2){
202            System.out.println("Ha ganado el jugador 2: " + j2);
203        }else{
204            System.out.println("Empate, el tablero esta lleno");
205        }
206        
207    }

209    public void iniciarPartida(Scanner entrada){
210
211        System.out.println("----CUATRO EN RAYA----");
212        System.out.println("jugador 1 = x");
213        System.out.println("jugador 2 = O");
214
215
216        while(!this.finPartida()){
217            System.out.println("---- jugada ----");
218            this.mostrarTabla();
219            System.out.println("");
220            this.mostrarTurno();
221
222            boolean entradaValida = false;
223            
224            while(!entradaValida){
225                System.out.println("Introduce la columna de 0-6: ");
226                if(!entrada.hasNextInt()){
227                    System.out.println("Error, Tiene que ser un numero");
228                    entrada.next();
229                }else{
230                    int columna = entrada.nextInt();
231
232                    if(columna < 0 || columna > 6){
233                        System.out.println("Columna invalida");
234                        entradaValida = false;
235                    }else{
236                        this.insertarEn(columna);
237                        entradaValida = true;
238                    }
239                }
240            }
241
242            if(this.finPartida()){
243                break;
244            }
245            this.cambioTurno();
246        }
247        this.mostrarTabla();
248        System.out.println("");
249        this.ganador();
250    }
